id = identifiant du tweet

timestamp_minute/hour/day/month/year = pour décrire la date et l'heure

geo_lat/lng = latitude et longitude (peut-etre null)

place_id = identifiant du district

text = ensemble des références et hashtags

user_id = identifiant de la personne qui Tweet

username = pseudo de la personne qui Tweet

fullname = nom complet de la personne qui Tweet

account_created_at = identifiant du lieu de création du compte de la personne qui Tweet

is_certified = certifié si compte officiel ou identité prouvée

followers_count = nbr de personnes qui suivent la personne qui Tweet

friends_count = nbr d'amis de la personne qui Tweet

tweets_count = nbr de Tweet deja envoyé

user_location = lieu du Tweet
